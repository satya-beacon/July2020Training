CREATE database eCommerceDb

USE eCommerceDb


CREATE TABLE Category(
ID INT IDENTITY PRIMARY KEY,
Name NVARCHAR(100),
[Description] NVARCHAR(MAX))



CREATE TABLE Users(
ID INT IDENTITY PRIMARY KEY,
FirstName NVARCHAR(100),
LastName NVARCHAR(100),
Email NVARCHAR(100),
Username NVARCHAR(100),
Password NVARCHAR(100))
